⚙️ Anatomy of settings.json

{
  "permissions": {
    "allow": ["Read", "Bash(git *)", "Bash(npm run *)"],
    "deny":  ["Bash(rm -rf *)"]
  },
  "hooks": {
    "PostToolUse": [
      { "matcher": "Write|Edit",
        "hooks": [{ "type": "command", "command": "npm run lint" }] }
    ]
  },
  "mcpServers": {
    "github": {
      "command": "npx",
      "args": ["-y", "@modelcontextprotocol/server-github"],
      "env": { "GITHUB_TOKEN": "ghp_..." }
    }
  }
}

LevelFileOverrides…
